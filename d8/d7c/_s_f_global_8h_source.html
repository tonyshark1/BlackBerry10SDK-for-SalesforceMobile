40 /*
41  * Macros for logging
42  * We don't use QT_NO_DEBUG_OUTPUT or QT_NO_WARNING_OUTPUT because we want the SDK to have
43  * separated log level from the application.
44  * Use SF_NO_DEBUG_OUTPUT and SF_NO_WARNING_OUTPUT to control SDK logging level.
45  *
46  * Note 1: QT macros overrides SF logging macros. i.e. if you have QT_NO_WARNING_OUTPUT defined,
47  * There will be no warning logs no matter if you defined SF_NO_WARNING_OUTPUT.
48  * Note 2: SF_NO_WARNING_OUTPUT implies SF_NO_DEBUG_OUTPUT
49  *
50  * e.g
51  * #define SF_NO_DEBUG_OUTPUT
52  * #define SF_NO_WARNING_OUTPUT
53  */
